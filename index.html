Material DateTimePicker (พ.ศ.)

วันที่ DD/MM/YYYY

$('#date').bootstrapMaterialDatePicker({
    format: 'DD/MM/YYYY',
    time: false
    okText: 'ตกลง'
    cancelText: 'ยกเลิก'
    clearText: 'เคลียร์'
    clearButton: true
});

เวลา HH:mm

$('#time').bootstrapMaterialDatePicker({
    format: 'HH:mm',
    date: false,
    okText: 'ตกลง'
    cancelText: 'ยกเลิก'
    clearText: 'เคลียร์'
});

วันที่และเวลา DD/MM/YYYY HH:mm

$('#date-format').bootstrapMaterialDatePicker({
    format: 'DD/MM/YYYY HH:mm',
    okText: 'ตกลง'
    cancelText: 'ยกเลิก'
    clearText: 'เคลียร์'
});
